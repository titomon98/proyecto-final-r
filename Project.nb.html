# Definimos los umbrales primero para integrarlos fácilmente después en las condicionales, será más fácil comprender el código así
media_productividad  = mean(proyectos$productividad, na.rm = TRUE)
sd_productividad     = sd(proyectos$productividad, na.rm = TRUE)
media_presupuesto    = mean(proyectos$presupuesto_usd, na.rm = TRUE)

# Realizamos la función para clasificar según los parámetors indicados, usando las categorías del ejemplo
clasificar_proyecto = function(productividad, presupuesto_usd, impacto) {
  if (impacto == "Alto" && productividad > media_productividad) {
    "Proyecto estratégico"
  } else if (presupuesto_usd > media_presupuesto && productividad < media_productividad) {
    "Riesgo e ineficiencia"
  } else if (productividad >= (media_productividad - sd_productividad) &&
             productividad <= (media_productividad + sd_productividad)) {
    "Proyecto estándar"
  } else {
    "Proyecto emergente"
  }
}

# Aplicamos al final la función a nuestros proyectos uno a uno
proyectos = proyectos %>%
  rowwise() %>%
  mutate(categoria_proyecto = clasificar_proyecto(productividad, presupuesto_usd, impacto)) %>%
  ungroup()

# Función para obtener el índice de eficiencia 
calcular_eficiencia = function(publicaciones, patentes, presupuesto_usd) {
  (publicaciones + patentes) / presupuesto_usd
}

# Función para obtener categoría estratégica según impacto y productividad, dado que no hay una regla definida
# Por lo tanto utilizaremos una simple condicional siendo que si el impacto es 'Alto' y su promedio de productividad está por encima del promedio
# Será estratégico, caso contrario no.

# Como dato, podríamos reutilizar la función que declaramos en la parte V, clasificar_proyecto, pero entran en juego otros parámetros extras
clasificar_impacto_estrategico = function(df) {
  df %>%
    mutate(
      categoria_estrategica = ifelse(
        impacto == "Alto" & productividad > mean(productividad, na.rm = TRUE),
        "Estratégico",
        "No estratégico"
      )
    )
}

# La última función para realizar un resumen estadístico de una variable numérica
resumen_estadistico = function(variable) {
  list(
    media             = mean(variable, na.rm = TRUE),
    mediana           = median(variable, na.rm = TRUE),
    desviacion_estandar = sd(variable, na.rm = TRUE),
    percentil_25      = quantile(variable, 0.25, na.rm = TRUE),
    percentil_75      = quantile(variable, 0.75, na.rm = TRUE),
    percentil_90      = quantile(variable, 0.90, na.rm = TRUE)
  )
}
